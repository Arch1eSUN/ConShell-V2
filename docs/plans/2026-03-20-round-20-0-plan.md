export class ChildSession {
    constructor(public name: string, private config: { budget: number }) {}
    getBudget() { return this.config.budget; }
}

// packages/core/src/orchestration/session-registry.ts
import { ChildSession } from './child-session';

export class SessionRegistry {
    private sessions: ChildSession[] = [];
    register(session: ChildSession) { this.sessions.push(session); }
    getActiveSessions() { return this.sessions; }
}
